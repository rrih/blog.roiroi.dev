GoでMySQLと接続

func dbConn() (db *sql.DB) {
	err := godotenv.Load(fmt.Sprintf("./%s.env", os.Getenv("GO_ENV")))
	if err != nil {
		fmt.Println("env err")
	}
	dbDriver := "mysql"
	dbUser := os.Getenv("DB_USER")
	dbPass := os.Getenv("DB_PASSWORD")
	dbName := "gosrc"
	db, openErr := sql.Open(dbDriver, dbUser+":"+dbPass+"@/"+dbName)
	if openErr != nil {
		log.Fatal(openErr.Error())
	}
	return db
}
